-- ^ -> representa o ctrl
-- M -> representa o alt
-- ^G -> ctrl + g

alt + \ -> começo do texto
alt + / -> final do texto

HOME -> inicio da linha
END -> final da linha

ctrl + shift + _ -> informa a linha e coluna onde ir.(navegar)
